        function loadRemoteData() {

            const inputUrl = document.getElementById('sheet-url-input').value.trim();

            if(!inputUrl) return alert("Please specify a valid published CSV URL first.");

            

            document.getElementById('sync-status').innerText = "Syncing...";

            Papa.parse(inputUrl, {

                download: true,

                header: true,

                dynamicTyping: true,

                skipEmptyLines: true,

                success: function(results) {

                    if(results.data && results.data.length > 0) {

                        document.getElementById('sync-status').className = "px-3 py-1 text-xs rounded-full bg-emerald-600 text-white";

                        document.getElementById('sync-status').innerText = "Synced Live";

                        renderGenerationalTree(results.data);

                    }

                },

                error: function(err) {

                    alert("Parsing error: " + err.message);

                    document.getElementById('sync-status').innerText = "Failed";

                }

            });

        }


        function renderGenerationalTree(flatData) {

            // Clear prior canvas elements cleanly before redraws

            document.getElementById('tree-container').innerHTML = '';


            // Convert raw flat tabular items into an organized parent-child dictionary for D3 usage

            try {

                const dataStratified = d3.stratify()

                    .id(d => d.GroupID)

                    .parentId(d => d.ParentID === 'Gen0' ? null : d.ParentID)(flatData);


                const margin = {top: 20, right: 90, bottom: 30, left: 90};

                const width = 700 - margin.left - margin.right;

                const height = 500 - margin.top - margin.bottom;


                const svg = d3.select("#tree-container").append("svg")

                    .attr("width", width + margin.left + margin.right)

                    .attr("height", height + margin.top + margin.bottom)

                    .append("g")

                    .attr("transform", "translate(" + margin.left + "," + margin.top + ")");


                const treemap = d3.tree().size([height, width]);

                let nodes = d3.hierarchy(dataStratified, d => d.children);

                nodes = treemap(nodes);


                // Add links joining generation layers together

                svg.selectAll(".link")

                    .data(nodes.descendants().slice(1))

                    .enter().append("path")

                    .attr("class", "link")

                    .attr("d", d => "M" + d.y + "," + d.x + "C" + (d.y + d.parent.y) / 2 + "," + d.x + " " + (d.y + d.parent.y) / 2 + "," + d.parent.x + " " + d.parent.y + "," + d.parent.x);


                // Instantiation loop for individual node elements

                const node = svg.selectAll(".node")

                    .data(nodes.descendants())

                    .enter().append("g")

                    .attr("class", d => "node" + (d.children ? " node--internal" : " node--leaf"))

                    .attr("transform", d => "translate(" + d.y + "," + d.x + ")")

                    .on("click", (event, d) => populateInspector(d.data.data));


                // Color code node borders to match Four Fields framework lifecycle status values

                node.append("circle")

                    .attr("r", 8)

                    .style("stroke", d => {

                        const s = d.data.data.Status;

                        if(s === 'Multiplying') return '#a855f7'; // Purple

                        if(s === 'Harvesting') return '#10b981'; // Emerald Green

                        if(s === 'Growing') return '#3b82f6'; // Blue

                        return '#eab308'; // Amber Seeding status

                    });


                node.append("text")

                    .attr("dy", ".35em")

                    .attr("x", d => d.children ? -13 : 13)

                    .style("text-anchor", d => d.children ? "end" : "start")

                    .text(d => d.data.data.GroupName);


            } catch (treeError) {

                document.getElementById('tree-container').innerHTML = `

                    <div class="text-red-500 text-sm p-4 border border-dashed border-red-300 rounded text-center">

                        <strong>Mapping Architecture Error:</strong><br>

                        Could not build tree. Check that your <code>ParentID</code> codes point precisely to existing <code>GroupID</code> targets.

                    </div>`;

                console.error(treeError);

            }

        }


        function populateInspector(group) {

            document.getElementById('inspector-content').innerHTML = `

                <div class="space-y-2 mt-1">

                    <div class="text-lg font-bold text-blue-900">${group.GroupName}</div>

                    <hr class="border-gray-100">

                    <div><span class="text-xs font-bold uppercase text-gray-400 block">Unique ID</span> <code class="bg-gray-100 px-1 py-0.5 rounded text-xs text-red-600">${group.GroupID}</code></div>

                    <div><span class="text-xs font-bold uppercase text-gray-400 block">Parent Root</span> <code class="bg-gray-100 px-1 py-0.5 rounded text-xs">${group.ParentID}</code></div>

                    <div><span class="text-xs font-bold uppercase text-gray-400 block">Field Facilitator</span> <span class="font-medium text-gray-800">${group.LeaderName}</span></div>

                    <div><span class="text-xs font-bold uppercase text-gray-400 block">Attendance Group Size</span> <span class="font-medium text-gray-800">${group.TotalAttendance} regular attendees</span></div>

                    <div><span class="text-xs font-bold uppercase text-gray-400 block">Four Fields Stage</span> <span class="inline-block mt-0.5 px-2 py-0.5 text-xs rounded font-semibold bg-gray-100 text-gray-700 border">${group.Status}</span></div>

                </div>`;

        }
